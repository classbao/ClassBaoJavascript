1，编码Html源代码：

        
var htmlStr = "<p><a href="./HtmlTag.html">HTML标签</a></p>";
var encodingHTML = CBJS.encodingHTML(htmlStr);
console.log("encodingHTML", encodingHTML);
项目值
编码Html源代码

2，解码Html源代码：

        
var decodingHTML = CBJS.decodingHTML(encodingHTML);
console.log("decodingHTML", decodingHTML);
项目值
解码Html源代码

3，移除字符串中的Html标签：

        
var removeHtmlTag = CBJS.removeHtmlTag(htmlStr);
console.log("removeHtmlTag", removeHtmlTag);
项目值
移除字符串中的Html标签

4，Unicode编码字符串：

        
var encodingUnicode = CBJS.encodingUnicode("我爱中国");
console.log("encodingUnicode", encodingUnicode);
项目操作值
Unicode编码字符串

5，Unicode解码字符串：

        
var decodingUnicode = CBJS.decodingUnicode("\u6211\u7231\u4e2d\u56fd");
console.log("decodingUnicode", decodingUnicode);
项目值
Unicode解码字符串

6，转义单引号（基于Sql规范）：

        
var escapeSingleQuotesForSQL = CBJS.escapeSingleQuotesForSQL("我是一个单引号'结束");

项目值
转义单引号（基于Sql规范）

var escapeSingleQuotesForJS = CBJS.escapeSingleQuotesForJS("我是一个单引号'结束");

转义单引号（基于Sql规范）

8，字符串长度（字节长度，真实字节长度）：

        
var realLength = CBJS.realLength("中国China");
console.log("realLength", realLength);
//output: 9
项目值
字符串长度（字节长度，真实字节长度）

9，按字节数截取字符串：

        
var subString = CBJS.subString("中国China", 6, "...");
console.log("subString", subString);
//output: 中国Ch...

参数解析：CBJS.subString(str, len, suffix)

> str:原字符串

> len:要截取长度（字节数，从0开始）

> suffix:超出长度被截掉的字符串显示后缀

项目值
按字节数截取字符串

10，移除URL末尾的空格、格式符号、“&”、“?”：

        
var clearURLEnd = CBJS.clearURLEnd("./HtmlTag.html \n?& ");
console.log("clearURLEnd", clearURLEnd);
//output: ./HtmlTag.html
项目值
移除URL末尾的空格、格式符号、“&”、“?”

11，获取区间内随机数：

        
var getRandom = CBJS.getRandom(1, 9);
console.log("getRandom", getRandom);

参数解析：CBJS.getRandom(minValue, maxValue)

> minValue:最小值(可以随机到该值)

> maxValue:最大值(可以随机到该值)

> 返回值:获取区间内随机数（可以取到最大值和最小值，均衡分布）

项目值
获取区间内随机数
